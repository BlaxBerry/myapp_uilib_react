## 受控组件写法

在使用 BaseInput 的组件内使用 state 管理输入值

<div className="my-code-block">
  <Story of={Stories.ControlledExample} />
</div>

```tsx
import React from "react";
import { BaseInput } from "myapp_uilib_react";

function App() {
  const [value, setValue] = React.useState<string>("");
  const onChange: React.ChangeEventHandler<HTMLInputElement> =
    React.useCallback((e) => setValue(e.target.value), []);

  return (
    <>
      <BaseInput value={value} onChange={onChange} />
      <p>{value}</p>
    </>
  );
}
```
